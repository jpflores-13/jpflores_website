Air quality 

with(airquality, boxplot(Temp ~ Month))

with(airquality, plot(Ozone ~ Temp))

mlev <- levels(with(airquality, as.factor(Month)))
with(airquality, plot(Ozone ~ Temp, 
                      pch = as.numeric(mlev), 
                      col = mlev))
